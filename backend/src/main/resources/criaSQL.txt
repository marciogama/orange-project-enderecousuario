CREATE TABLE tb_endereco (id integer generated by default as identity,logradouro varchar(120),numero varchar(30),complemento varchar(50),bairro varchar(50),cidade varchar(50),estado varchar(2),cep varchar(8));
CREATE TABLE tb_usuario (id integer generated by default as identity,nome varchar(120) NOT NULL,email varchar(120) UNIQUE NOT NULL,cpf varchar(11) UNIQUE NOT NULL,datanascimento date);
CREATE TABLE tb_usuario_endereco (usuario_id integer NOT NULL,endereco_id integer NOT NULL);
